export default {
  post(url, data) {
    return axios({
      method: "post",
      baseURL: BASE_URL.BASE_URL_PREFIX,
      url,
      data: qs.stringify(data),
      timeout: HTTP_TIMEOUT,
    })
      .then(response => {
        return response;
      })
      .catch(error => {
        return error;
      });
  },
  get(url, data) {
    return axios({
      method: "get",
      baseURL: BASE_URL.BASE_URL_PREFIX,
      url,
      params: data, // get 请求时带的参数
      timeout: HTTP_TIMEOUT,
    })
      .then(response => {
        return response;
      })
      .catch(error => {
        return error;
      });
  }
};
